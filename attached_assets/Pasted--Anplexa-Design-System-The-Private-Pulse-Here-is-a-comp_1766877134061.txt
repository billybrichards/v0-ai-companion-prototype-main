### **Component Implementation Preview**

The following Python code generates a visual "UI Kit" sheet based directly on the JSON data above. This allows you to verify how the colors, typography, and component styles come together visually before writing the CSS.

```python
import matplotlib.pyplot as plt
import matplotlib.patches as patches
import matplotlib.font_manager as fm

def draw_style_guide():
    # --- 1. SETUP & CONFIG ---
    fig, ax = plt.subplots(figsize=(16, 12))
    fig.patch.set_facecolor('#0a0a0a') # Deep background for presentation
    ax.set_facecolor('#121212') # "Midnight Void" app background
    ax.set_xlim(0, 1600)
    ax.set_ylim(0, 1200)
    ax.invert_yaxis() # Top-down coordinates
    ax.axis('off')

    # Font handling (Simulating Montserrat/Inter via system fonts)
    title_font = fm.FontProperties(family='sans-serif', weight='light', size=28)
    header_font = fm.FontProperties(family='sans-serif', weight='bold', size=18)
    body_font = fm.FontProperties(family='sans-serif', weight='normal', size=14)
    mono_font = fm.FontProperties(family='monospace', size=10)

    # --- 2. COLOR PALETTE SECTION ---
    def draw_swatch(x, y, color, name, hex_code):
        rect = patches.Rectangle((x, y), 120, 80, linewidth=0, facecolor=color)
        ax.add_patch(rect)
        ax.text(x, y + 100, name, color='#E0E1DD', fontproperties=body_font, fontsize=11)
        ax.text(x, y + 120, hex_code, color='#777', fontproperties=mono_font)

    ax.text(50, 60, "ANPLEXA DESIGN SYSTEM v1.0", color='#7B2CBF', fontproperties=title_font)
    
    ax.text(50, 140, "01 COLORS", color='#415A77', fontproperties=header_font)
    draw_swatch(50, 170, '#7B2CBF', 'Anplexa Purple', '#7B2CBF')
    draw_swatch(200, 170, '#121212', 'Midnight Void', '#121212')
    draw_swatch(350, 170, '#E0E1DD', 'Ghost Silver', '#E0E1DD')
    draw_swatch(500, 170, '#415A77', 'Secure Slate', '#415A77')

    # --- 3. TYPOGRAPHY SECTION ---
    ax.text(50, 320, "02 TYPOGRAPHY", color='#415A77', fontproperties=header_font)
    
    ax.text(50, 370, "Display H1 - Montserrat Light", color='#E0E1DD', fontsize=32, fontweight='light')
    ax.text(50, 420, "Heading H2 - Montserrat Regular", color='#E0E1DD', fontsize=24, fontweight='normal')
    ax.text(50, 460, "Body Regular - Inter 16px. The quick brown fox jumps over the lazy dog.", color='#A0A19D', fontsize=16)
    ax.text(50, 490, "Caption / Mono - Fira Code", color='#555', fontfamily='monospace', fontsize=12)

    # --- 4. COMPONENTS SECTION ---
    ax.text(700, 140, "03 COMPONENTS", color='#415A77', fontproperties=header_font)

    # -- Primary Button --
    ax.text(700, 180, "Primary Button", color='#555', fontsize=10)
    btn_rect = patches.FancyBboxPatch((700, 200), 200, 50, boxstyle="round,pad=0,rounding_size=8", facecolor='#7B2CBF', edgecolor='none')
    ax.add_patch(btn_rect)
    ax.text(800, 230, "INITIALIZE SESSION", color='white', ha='center', va='center', fontweight='bold', fontsize=11, letter_spacing=1)

    # -- Input Field (Active) --
    ax.text(700, 280, "Input Field (Focus)", color='#555', fontsize=10)
    input_rect = patches.FancyBboxPatch((700, 300), 300, 50, boxstyle="round,pad=0,rounding_size=8", facecolor='#121212', edgecolor='#7B2CBF', linewidth=1.5)
    ax.add_patch(input_rect)
    # Glow effect for focus
    glow_rect = patches.FancyBboxPatch((695, 295), 310, 60, boxstyle="round,pad=0,rounding_size=10", facecolor='#7B2CBF', alpha=0.15)
    ax.add_patch(glow_rect)
    ax.text(720, 325, "Hello, Anplexa...", color='#E0E1DD', va='center', fontsize=12)
    # Blinking cursor
    ax.plot([850, 850], [315, 335], color='#7B2CBF', linewidth=2)

    # -- Chat UI Cards --
    ax.text(700, 400, "Chat Interface", color='#555', fontsize=10)
    
    # User Bubble
    user_bubble = patches.FancyBboxPatch((1000, 430), 250, 60, boxstyle="round,pad=0,rounding_size=12", facecolor='#1A1A1A')
    ax.add_patch(user_bubble)
    ax.text(1230, 460, "Can we talk privately?", color='#A0A19D', va='center', ha='right', fontsize=12)
    
    # AI Bubble (Active)
    ai_bubble = patches.FancyBboxPatch((700, 510), 280, 80, boxstyle="round,pad=0,rounding_size=12", facecolor='#252525')
    ax.add_patch(ai_bubble)
    # Left accent border
    ax.plot([700, 700], [525, 575], color='#7B2CBF', linewidth=4) 
    ax.text(720, 540, "Connection secure.\nI'm listening. Tell me everything.", color='#E0E1DD', va='center', fontsize=12)

    # --- 5. LOGO USAGE ---
    ax.text(50, 600, "04 LOGO USAGE", color='#415A77', fontproperties=header_font)
    
    # Background for Logo
    logo_bg = patches.Rectangle((50, 640), 120, 120, facecolor='#000', edgecolor='#222', linewidth=1)
    ax.add_patch(logo_bg)
    
    # Draw Minimal Logo (Concept A - Pulse)
    import numpy as np
    x = np.linspace(70, 150, 50)
    y1 = 700 + 10 * np.sin((x-70) * np.pi / 20)
    y2 = 700 - 10 * np.sin((x-70) * np.pi / 20)
    ax.plot(x, y1, color='#E0E1DD', linewidth=2)
    ax.plot(x, y2, color='#7B2CBF', linewidth=2)
    ax.text(110, 740, "Icon Only", color='#555', fontsize=9, ha='center')

    # Security Badge
    ax.text(250, 640, "Trust Indicator", color='#555', fontsize=10)
    badge = patches.FancyBboxPatch((250, 660), 200, 40, boxstyle="round,pad=0,rounding_size=20", facecolor='none', edgecolor='#415A77', linewidth=1)
    ax.add_patch(badge)
    ax.text(350, 680, "ðŸ”’  End-to-End Encrypted", color='#415A77', ha='center', va='center', fontsize=10)

    plt.tight_layout()
    plt.show()

draw_style_guide()
```
